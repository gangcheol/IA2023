2. 평행

- 두 직선이 평행인 경우 1, 아니면 0

dots = [[1, 4], [9, 2], [3, 8], [11, 6]]

sol

- 평행인 2개의 직선을 그려보면 왜 \(x\)값을 기준으로 정렬해서 풀어야하는지 금방 이해할 수 있다.

def solution (dots) :
    print("정렬 전 : ", dots)
    dots.sort()
    print("정렬 후 : ", dots)
    [x1, y1], [x2, y2] = dots[:2]
    [x3, y3], [x4, y4] = dots[2:]
    
    a1 = (y1-y2)/(x1-x2)
    a2 = (y3-y4)/(x3-x4)
    
    if a1==a2 :
        return 1
    else : 
        return 0
solution(dots)

정렬 후 :  [[1, 4], [3, 8], [9, 2], [11, 6]]

sol2

\[\text {기울기 구하는 공식} = \frac{(y_1-y_2)}{(x_1-x_2)}\]

dots
[[1, 4], [3, 8], [9, 2], [11, 6]]
def solution(dots):
    [[x1, y1], [x2, y2], [x3, y3], [x4, y4]]=dots
    answer1 = ((y1-y2)*(x3-x4) == (y3-y4)*(x1-x2))
    answer2 = ((y1-y3)*(x2-x4) == (y2-y4)*(x1-x3))
    answer3 = ((y1-y4)*(x2-x3) == (y2-y3)*(x1-x4))
    return 1 if answer1 or answer2 or answer3 else 0
solution(dots)
1

t = "3141592"
p = "271"
def solution(t,p) : 
    n = len(p)
    return len([int(t[i:i+n]) for i in range(len(t)-n+1) if int(t[i:i+n]) <= int(p) ])

def solution(p,c) :
    p.sort()
    c.sort()
    for i in range(len(c)) :
        if p[i] != c[i] :
            return p[i]
    return p[-1]

5. 겹치는 선분의 길이

예비학습

lines = [[0, 5], [3, 9], [1, 10]]
lines
[[0, 5], [3, 9], [1, 10]]
sets = [set(range(l[0],l[1])) for l in lines]
sets
[{0, 1, 2, 3, 4}, {3, 4, 5, 6, 7, 8}, {1, 2, 3, 4, 5, 6, 7, 8, 9}]

\[\text{sets[0]} \cap \text {sets[1]}= \text{{3, 4}}\]

(sets[0] & sets[1])
{3, 4}
len((sets[0] & sets[1]) | (sets[0] & sets[2]) | (sets[1] & sets[2]))
8

sol

def solution(lines):
    sets = [set(range(min(l), max(l))) for l in lines]
    return len(sets[0] & sets[1] | sets[0] & sets[2] | sets[1] & sets[2])
solution(lines)
8
