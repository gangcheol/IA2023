1. 저주의 숫자 3

- 정수 n이 주어질 때 위와 같은 규칙을 따르는 숫지를 return 하는 함수를 작성하라

sol

  1. 임의의 숫자 n을 입력 받았을 때, 1부터 n 번째 까지의 상황을 기록하는 변수(r)가 필요
r = 0
for i in range(1, n+1) :
    r += 1

2. 변수 r이 3의 배수 이거나 3을 포함하고 있는 string일 경우 r+1을 수행

n=15
r = 0
for i in range(1, n+1) :
    r +=1
    while (r % 3 == 0) or ('3' in str(r)) : ## 30,31,32... 이럴경우 for문에서 계속 반복해야 하므로
        r+=1
r
25

3. 함수로 작성

def solution(n) : 
        r = 0
        for i in range(1, n+1) :
            r +=1
            while (r % 3 == 0) or ('3' in str(r)) : ## 30,31,32... 이럴경우 for문에서 계속 반복해야 하므로
                r+=1
        return r
solution(5)
7

l = l.split()

result = []
for i in lst :

    if l[1] == "-" :

- 함수 작성

sol1

lst =["19 - 6 = 13", "5 + 66 = 71", "5 - 15 = 63", "3 - 1 = 2"]

def solution(lst) : 
    result = []
    for i in lst :
        l,r=i.split("=")
        l = l.split()
        if l[1] == "-" :
            temp = int(l[0]) - int(l[-1]) == int(r)
        else :
            temp = int(l[0]) + int(l[-1]) == int(r)

        result.append(temp)
    print(result)
    result = ["O" if i == True else "X" for i in result]
    return result
solution(lst)

sol2 (쩌는거…)

def valid(equation):
    equation = equation.replace('=', '==')
    return eval(equation)

def solution(equations):
    return ["O" if valid(equation) else "X" for equation in equations]
lst =["19 - 6 = 13", "5 + 66 = 71", "5 - 15 = 63", "3 - 1 = 2"]
solution(lst)
['O', 'O', 'X', 'O']

1. 정수를 나선형으로 배치하기

sol

def solution(n):
    if n == 1:
        return [[1]]
    
    answer = [[0 for j in range(n)] for i in range(n)] 
    x = 0
    y = 0
    dir = 'r'
    for i in range(n*n):
        answer[x][y] = i + 1
        if dir == 'r':
            y += 1
            if y == n-1 or answer[x][y+1] != 0: 
                dir = 'd'
        elif dir == 'd':
            x += 1
            if x == n-1 or answer[x+1][y] != 0:
                dir = 'l'
        elif dir == 'l':
            y -= 1
            if y == 0 or answer[x][y-1] != 0:
                dir = 'u'
        elif dir == 'u':
            x -= 1
            if x == n-1 or answer[x-1][y] != 0:
                dir = 'r'        
    return answer
solution(4)
[[1, 2, 3, 4], [12, 13, 14, 5], [11, 16, 15, 6], [10, 9, 8, 7]]
